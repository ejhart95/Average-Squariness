import math
test_length = 10000000
total_squariness = 0

squariness = [0]*test_length

for i in range(test_length):
    j = i +1
    m = math.floor(math.sqrt(j))
    for p in range(m):
        if j % (p+1) == 0:
            # This means (j) = p*q, which can be rearranged as q = (j)/p. Therefore p/q = p/((j)/p) = (p^2)/j.
            squariness[i] = ((p+1)**2)/(j)

total_squariness = sum(squariness)

average_squariness = total_squariness/test_length

print(average_squariness)
